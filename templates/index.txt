                const searchButton = document.getElementById('searchButton');
                const resultsContainer = document.getElementById('resultsContainer');
                const bankStatementFormContainer = document.getElementById('bankStatementFormContainer');

                searchButton.addEventListener('click', function() {
                    var searchWords = document.getElementById('search_words').value;
                    if (searchWords) {
                        document.getElementById('searchForm').action = '/?search_words=' + searchWords;
                        document.getElementById('searchForm').submit();
                        resultsContainer.style.display = 'block'; // Show results container
                        bankStatementFormContainer.style.display = 'none'; // Hide other form
                    } else {
                        alert("Please enter search words."); // Alert if searchWords is empty
                    }
                });

                // Hide results container and other form initially
                resultsContainer.style.display = 'none';
                bankStatementFormContainer.style.display = 'none';
